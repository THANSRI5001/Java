import java.util.Scanner;
public class Main{
	public static void main(String[] args)
	{
		Scanner sc = new Scanner(System.in);
		int input1 = sc.nextInt();
		int input2 = sc.nextInt();
		int sum = 0;
		sum = input1+input2;
		if(input1==input2)
		{
			System.out.println(sum);
		}
		else
		{
			System.out.println(sum*2);
		}
		
		sc.close();
	}
}
